                            //get the values from the Page

                            function getValues(){
                                //get values from the page 
                                let startValue = document.getElementById("startValue").value;
                                let endValue = document.getElementById("endValue").value;
                                
                                //Validate input
                                //Parse into integer
                                startValue = parseInt(startValue);
                                endValue = parseInt(endValue);

                                if (Number.isInteger(startValue) && Number.isInteger(endValue)) {
                                    //call generateNumbers
                                    let numbers = generateNumbers(startValue, endValue);
                                    //call displayNumbers
                                    displayNumbers(numbers);     

                                }else{
                                    alert("You must enter integers.");
                                }
                            }

                            //generate numbers from startvalue to the endvalue
                            //logic function(s)
                            function generateNumbers(sValue, eValue){

                                let numbers = [];

                                //get all numbers from the start to the end
                                for (let i = sValue; i <= eValue; i++) {

                                    //this will execute until the i(index) gets the eValue
                                    numbers.push(i);
                                    
                                }

                                return numbers;

                            }

                            //display the numbers and mark even numbers bold
                            //display or view functions
                            function displayNumbers(numbers){

                                let templateRows="";

                                for (let index = 0; index < numbers.length; index++) {
                                    
                                    let className = "even";

                                    let number = numbers[index];

                                    if(number % 2 == 0){
                                        className = "even";
                                    }
                                    else{
                                        className = "odd";
                                    }
                                    //This line doesn not render correctly with Prism
                                    templateRows += `${number}`;
                                }

                                document.getElementById("results").innerHTML = templateRows;
                            }

The Evenly Matched app's code is written with HTML5, Bootstrap, CSS, and JavaScript to accomplish its functionality.

Overview

Evenly Matched takes numerical inputs from the user and identifies even numbers from the given range. The result is then elegantly displayed with even numbers emphasized in bold.

getValues()

This function acts as the controller, extracting the start and end values from the user interface. It ensures the provided inputs are valid integers before invoking subsequent functions to generate and display the sequence of numbers.

generateNumbers(sValue, eValue)

Designed to be a utility, this function accepts a starting and ending value, creating an array of numbers within that range. It's a pure logic function, returning the sequence for further processing.

displayNumbers(numbers)

With a sharp focus on presentation, this function handles the display logic. For each number in the provided array, it determines its parity and applies the "even" class for even numbers, making them stand out in bold. This visual distinction is achieved through CSS and ensures an intuitive user experience.
